Language Fundamentals - Solutions

These are suggested solutions for some of the exercises in this chapter. There are usually several ways to solve a problem and the solutions provided illustrate one approach.

  1. Temperature Conversion

    In Celsius:

    package solutions;

public class MainConversion {
    public static void main(String[] args) {
        double f451 = (451.0 - 32.0) * (5.0 / 9.0);
        System.out.println(f451);
    }
}

    In Kelvin:

    package solutions;

public class MainConversion {
    public static void main(String[] args) {
        double f451 = (451.0 - 32.0) * (5.0 / 9.0) - 273.15;
        System.out.println(f451);
    }
}

  2. Number Classifier

    package solutions;

import java.util.Random;

public class MainClassifier {
    public static void main(String[] args) {
        Random random = new Random();
        int number = random.nextInt(98)+1;
        System.out.print(number);
        if (number >= 67) {
            System.out.println(" is large");
        }
        else if (number >= 33) {
            System.out.println(" is medium");
        }
        else {
            System.out.println(" is small");
        }
    }
}

  3. Countdown

    First version with hard coded start value of 10.

    package solutions;

public class MainCountdown {
    public static void main(String[] args) {
        int start = 10;
        while (start >= 0) {
            System.out.println(start);
            --start;
        }
    }
}

    Second version with user supplied start value.

    package solutions;

import java.util.Scanner;

public class MainCountdown {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a start count? ");
        int start = scanner.nextInt();
        while (start >= 0) {
            System.out.println(start);
            --start;
        }
    }
}

  4. Times Tables

    package solutions;

public class MainTables {
    public static void main(String[] args) {
        for (int row=1; row<=12; ++row) {
            for (int col=1; col<=12; ++col) {
                System.out.print(row*col);
                System.out.print(' ');
            }
            System.out.println();
        }
    }
}
